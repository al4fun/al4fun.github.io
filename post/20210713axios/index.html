// 生成配置
// name为'axios'和'axios.min'
function generateConfig(name) {
  var compress = name.indexOf('min') > -1; // 是否进行代码压缩
  var config = {
    entry: './index.js', // 打包起点
    output: {
      path: __dirname + '/dist/',
      filename: name + '.js',
      sourceMapFilename: name + '.map',
      library: 'axios', // 库的名称，参考https://webpack.docschina.org/configuration/output/#outputlibrary
      libraryTarget: 'umd' // 打umd包
    },
    node: {
      process: false
    },
    devtool: 'source-map',
    mode: compress ? 'production' : 'development' // production打包模式下，webpack会自动进行代码压缩等操作
  };
  return config;
}

['axios', 'axios.min'].forEach(function (key) {
  config[key] = generateConfig(key);
});

module.exports = config;

// axios.js

// 创建axios实例
// axios既是函数，也是对象，既可以axios(config)，也可以axios.get(config)。很多流行的库都有类似的实现
function createInstance(defaultConfig) {
  // 创建context
  var context = new Axios(defaultConfig);
  // 创建instance，实际是Axios.prototype.request函数
  // 将Axios.prototype.request中的this设置为context
  var instance = bind(Axios.prototype.request, context);

  // 将Axios.prototype对象自有的属性和方法拷贝到instance中，方法的this指定为context
  utils.extend(instance, Axios.prototype, context);
  // 将context对象自有的属性和方法拷贝到instance中
  utils.extend(instance, context);

  return instance;
}

可见axios本质上就是request函数，只是将Axios实例的属性和方法拷贝到了axios中。

// Axios.js

// 构造函数
function Axios(instanceConfig) {
  this.defaults = instanceConfig;

  // 拦截器
  this.interceptors = {
    request: new InterceptorManager(),
    response: new InterceptorManager()
  };
}

// request函数
Axios.prototype.request = function request(config) {
  // 处理axios(url[,config])
  if (typeof config === 'string') {
    config = arguments[1] || {};
    config.url = arguments[0];
  } else {
    config = config || {};
  }

  // 合并config和默认配置
  config = mergeConfig(this.defaults, config);

  // 处理请求方式
  if (config.method) {
    config.method = config.method.toLowerCase();
  } else if (this.defaults.method) {
    config.method = this.defaults.method.toLowerCase();
  } else {
    config.method = 'get';
  }

  // 响应转换器(略)
  var transitional = config.transitional;
  if (transitional !== undefined) {
    validator.assertOptions(transitional, {
      silentJSONParsing: validators.transitional(validators.boolean, '1.0.0'),
      forcedJSONParsing: validators.transitional(validators.boolean, '1.0.0'),
      clarifyTimeoutError: validators.transitional(validators.boolean, '1.0.0')
    }, false);
  }

  // 将满足执行条件的请求拦截器加入到requestInterceptorChain中
  var requestInterceptorChain = [];
  var synchronousRequestInterceptors = true; // 是否全部请求拦截器都是synchronous的
  this.interceptors.request.forEach(function unshiftRequestInterceptors(interceptor) {
    if (typeof interceptor.runWhen === 'function' && interceptor.runWhen(config) === false) {
      return;
    }
    synchronousRequestInterceptors = synchronousRequestInterceptors && interceptor.synchronous;
    requestInterceptorChain.unshift(interceptor.fulfilled, interceptor.rejected);
  });
  // 将响应拦截器加入到responseInterceptorChain中
  var responseInterceptorChain = [];
  this.interceptors.response.forEach(function pushResponseInterceptors(interceptor) {
    responseInterceptorChain.push(interceptor.fulfilled, interceptor.rejected);
  });

  var promise;

  // 分支1：有异步执行的请求拦截器
  if (!synchronousRequestInterceptors) {
    // 创建执行链chain
    // dispatchRequest是真正执行网络请求的函数
    var chain = [dispatchRequest, undefined];

    // 将requestInterceptorChain和responseInterceptorChain分别连接到chain的首尾
    Array.prototype.unshift.apply(chain, requestInterceptorChain);
    chain.concat(responseInterceptorChain);

    // 遍历执行chain
    promise = Promise.resolve(config);
    while (chain.length) {
      promise = promise.then(chain.shift(), chain.shift());
    }

    return promise;
  }

  // 分支2：全部请求拦截器都是同步执行的(略)
  // 遍历执行请求拦截器
  var newConfig = config;
  while (requestInterceptorChain.length) {
    var onFulfilled = requestInterceptorChain.shift();
    var onRejected = requestInterceptorChain.shift();
    try {
      newConfig = onFulfilled(newConfig);
    } catch (error) {
      onRejected(error);
      break;
    }
  }
  // 执行dispatchRequest
  try {
    promise = dispatchRequest(newConfig);
  } catch (error) {
    return Promise.reject(error);
  }
  // 遍历执行响应拦截器
  while (responseInterceptorChain.length) {
    promise = promise.then(responseInterceptorChain.shift(), responseInterceptorChain.shift());
  }
  return promise;
};

function getDefaultAdapter() {
  var adapter;
  if (typeof XMLHttpRequest !== 'undefined') {
    // 浏览器环境，使用XHR adapter
    adapter = require('./adapters/xhr');
  } else if (typeof process !== 'undefined' && Object.prototype.toString.call(process) === '[object process]') {
    // node环境，使用HTTP adapter
    adapter = require('./adapters/http');
  }
  return adapter;
}

module.exports = function xhrAdapter(config) {
  return new Promise(function dispatchXhrRequest(resolve, reject) {
    ...

    var request = new XMLHttpRequest();
    var fullPath = buildFullPath(config.baseURL, config.url);
    request.open(config.method.toUpperCase(), buildURL(fullPath, config.params, config.paramsSerializer), true);

    // Handle browser request cancellation (as opposed to a manual cancellation)
    request.onabort = function handleAbort() {
      reject(createError('Request aborted', config, 'ECONNABORTED', request));
      request = null;
    };

    // Handle low level network errors
    request.onerror = function handleError() {
      reject(createError('Network Error', config, null, request));
      request = null;
    };

    ...

    // 处理cancelToken(用户取消请求)
    if (config.cancelToken) {
      config.cancelToken.promise.then(function onCanceled(cancel) {
        if (!request) return;
        request.abort();
        reject(cancel);
        request = null;
      });
    }

    // Send the request
    request.send(requestData);
  });
};
