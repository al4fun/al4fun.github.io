config.js主要用于配置和存储上报所需的一些基本信息，包括上报地址、app版本等。

监控上报功能：watchXXX.js

为方便用户自由选择，将监控上报功能按类别拆分成了watch-error、watch-web-vitals、watch-fps等多个模块。每个模块会对外暴露一个watchXXX函数，如watchError、watchWebVitals、watchFPS等，调用函数则启动相应的监控上报功能。

function init({ config = {}, watchers = [watchError, watchFPS, watchWebVitals] } = {}) {
    setConfig(config);

    // 遍历调用watcher
    // watcher可能是函数，也可能是包含func和params两个字段的对象
    watchers.forEach(watcher => {
        if (typeof watcher === 'function') {
            watcher();
        } else if (isObject(watcher) && typeof watcher.func === 'function' && isObject(watcher.params)) {
            watcher.func(watcher.params);
        }
    });

    initReport();
}

class BaseInfo {
    constructor({ type, subType }) {
        this.type = type;
        this.subType = subType;
        this.timestamp = new Date().valueOf();
        this.pageUrl = window.location.href;
    }
}
class ErrorInfo extends BaseInfo {
    constructor({ subType, message, stack }) {
        super({ type: 'ErrorInfo', subType: subType });
        this.message = message;
        this.stack = stack;
    }
}
class PerformanceInfo extends BaseInfo {
    constructor({ subType, value }) {
        super({ type: 'PerformanceInfo', subType: subType });
        this.value = value;
    }
}
class BehaviorInfo extends BaseInfo {
    constructor({ subType, value }) {
        super({ type: 'BehaviorInfo', subType: subType });
        this.value = value;
    }
}
class CustomInfo extends BaseInfo {
    constructor({ subType, value }) {
        super({ type: 'CustomInfo', subType: subType });
        this.value = value;
    }
}

// 延迟节流
function delayedThrotle(fn, delay) {
  let timer = null;

  return function () {
    if (timer != null) return;

    const context = this;
    const args = arguments;
    timer = setTimeout(function () {
      fn.apply(context, args);
      timer = null;
    }, delay);
  };
}

function batchUpload() {
    const batchCapacity = 10;
    const batchCount = Math.ceil(infos.length / batchCapacity);
    for (let i = 0; i < batchCount; i++) {
        const params = {
            userAgent: window.navigator.userAgent,
            appName: config.appName,
            appVersion: config.appVersion,
            userId: config.userId,
            infos: infos.slice(i * batchCapacity, Math.min((i + 1) * batchCapacity, infos.length)),
        }

        if (window.navigator.sendBeacon) {
            window.navigator.sendBeacon(config.reportUrl, JSON.stringify(params));
        } else {
            const url = `${config.reportUrl}/report.gif?params=${encodeURIComponent(JSON.stringify(params))}`;
            (new Image()).src = url;
        }
    }
    infos.length = 0;
}

如果浏览器支持sendBeacon，就使用sendBeacon进行上报。sendBeacon实际发出的是post请求，并且支持跨域。以下介绍摘自MDN：

navigator.sendBeacon()方法可用于通过HTTP将少量数据异步传输到服务器。

这个方法主要用于满足统计和诊断代码的需要，这些代码通常尝试在卸载（unload）文档之前向服务器发送数据。过早的发送数据可能导致错过收集数据的机会。

然而，对于开发者来说保证在文档卸载期间发送数据一直是一个困难。因为浏览器通常会忽略在 unload 事件处理器中产生的异步 XMLHttpRequest。为了解决这个问题， 统计和诊断代码通常要在 unload 或者 beforeunload 事件处理器中发起一个同步 XMLHttpRequest 来发送数据。同步的 XMLHttpRequest 迫使客户端延迟卸载文档，并使得下一个页面出现的更晚。下一个页面对于这种较差的载入表现无能为力。

这就是 sendBeacon() 方法存在的意义。使用 sendBeacon() 方法会使客户端在有机会时，异步地向服务器发送数据，同时不会延迟当前页面的卸载和下一页面的载入。

若浏览器不支持sendBeacon，则使用new Image()进行上报。相比传统GET/POST请求，通过new Image()向服务器提交数据可以避开跨域限制。

另一方面，因为浏览器和服务器对url的长度通常都有限制，为避免url长度超限，在batchUpload中还限制了单次上报的数据不超过10条。

最后，因为在throtledBatchUpload中使用了定时器，而当页面被卸载时(用户关闭标签页、刷新标签页、在标签页中打开另一网站)，定时器会被自动取消，为了避免infos中的数据因来不及上报而丢失，我们还需要在beforeunload事件被触发时，对infos中的数据进行一次立即上报：

function initReport() {
    // 在页面卸载之前，立即上报
    // 因为如果页面被卸载了，throtledBatchUpload内部的定时器也会被取消，造成上报数据丢失

import { report, ErrorInfo } from './report.js';

function watchError({ vueCls } = {}) {
    watchUnCaughtError();
    watchUnCaughtRejection();
    watchVueComponent(vueCls);
}

function watchUnCaughtError() {
    window.addEventListener('error', (error) => {
        report(format(error));
    }, true);
}

function watchUnCaughtRejection() {
    window.addEventListener('unhandledrejection', (error) => {
        report(format(error));
    });
}

function watchVueComponent(vueCls) {
    if (vueCls && vueCls.config) {
        const prev = vueCls.config.errorHandler;
        vueCls.config.errorHandler = function (error, vm, info) {
            console.error(error);

            report(format(error));

            if (typeof prev === 'function') {
                prev.call(this, error, vm, info);
            }
        }
    }
}

...

export default watchError;

通过window.addEventListener('error', callback, true)可以捕获到同步和异步的运行时错误，以及资源加载错误；通过window.addEventListener('unhandledrejection',callback)可以捕获到未被处理的promise的reject信息；而通过Vue.config.errorHandler则可以捕获到Vue组件中的各种错误。

import { report, PerformanceInfo } from './report.js';

// lowFPSThreshold = 20: FPS低于20视为lowFPS
// continuousThreshold = 3: 连续出现3个lowFPS，视为卡顿，进行上报
function watchFPS({ lowFPSThreshold = 20, continuousThreshold = 3 } = {}) {
    const continuousLowFPS = []; // 存储连续出现的lowFPS
    let frameCount = 0;
    let startTime = performance.now();

    function loop() {
        frameCount++;

        const now = performance.now();
        if (now - startTime >= 1000) {
            const fps = Math.round(frameCount * 1000 / (now - startTime));
            if (fps < lowFPSThreshold) {
                continuousLowFPS.push(fps);
                if (continuousLowFPS.length === continuousThreshold) {
                    // 达到连续lowFPS标准，进行上报
                    report(new PerformanceInfo({
                        subType: "FPS",
                        value: continuousLowFPS.reduce((sum, item) => sum + item) / continuousLowFPS.length
                    }));
                    continuousLowFPS.length = 0;
                }
            } else {
                continuousLowFPS.length = 0;
            }

            // 开始新一轮fps计算
            frameCount = 0;
            startTime = now;
        }

        window.requestAnimationFrame(loop);
    }

    loop();
}

export default watchFPS;

通过 requestAnimationFrame 来进行帧率计数。当连续 continuousThreshold 次帧率小于 lowFPSThreshold 时，就认为页面发生了卡顿，需要进行上报。

Google性能指标监控：watch-web-vitals.js

Web Vitals，即Web核心指标，是由google发起的一项新计划，旨在提供一份有关各种质量信号的统一指南。

Web核心指标的构成会随着时间的推移而发展。当前构成侧重于用户体验的三个方面：

  • LCP: Largest Contentful Paint，最大内容绘制，即从页面加载开始到页面可视区域内最大文本块或图像元素完成渲染的时间。衡量页面加载性能。
  • FID: First Input Delay，首次输入延迟，即从用户第一次与页面交互（如点击链接、点击按钮）到浏览器实际能够开始响应该交互的时间。衡量页面交互性能。
  • CLS: Cumulative Layout Shift，累积布局偏移，即页面的整个生命周期中所有意外布局偏移的累积分数。衡量页面视觉稳定性。

尽管google对每种指标都给出了详细的测量和计算方法，但因为要考虑和处理的情况十分复杂，google还是建议开发者直接使用其推出的web-vitals库。使用方法非常简单，详细参考: https://web.dev/vitals 。

import { getLCP, getFID, getCLS } from 'web-vitals';
import { report, PerformanceInfo } from './report.js';

const queue = new Set();

function addToQueue(metric) {
    queue.add(format(metric));
}

function flushQueue() {
    if (queue.size > 0) {
        report([...queue]);
        queue.clear();
    }
}

function watchWebVitals() {
    getCLS(addToQueue);
    getFID(addToQueue);
    getLCP(addToQueue);

    // 页面不可见
    document.addEventListener('visibilitychange', () => {
        if (document.visibilityState === 'hidden') {
            flushQueue();
        }
    });
    // 标签页关闭、刷新、打开另一个页面
    // 上述三种情况并不会触发visibilitychange
    window.addEventListener('beforeunload', () => {
        flushQueue();
    });
}

function format(metric) {
    return new PerformanceInfo({
        subType: metric.name,
        value: metric.value
    });
}

export default watchWebVitals;

按照google推荐的做法，使用一个set对web vitals数据进行缓存，并在页面不可见时统一上报。

此外，为避免标签页关闭、刷新、打开另一个页面时数据丢失，在页面卸载之前也要进行上报。

Vue页面加载渲染时间监控：watch-vue-load.js

参考文章：https://mp.weixin.qq.com/s/O2ANZE1vYG-DT6eCB_hf4A ，计算Vue页面加载渲染时间的思路如下：

  • 监听 router.beforeEach()，在回调函数中将当前时间记为渲染开始时间。
  • 利用 Vue.mixin() 对所有组件的 mounted() 注入一段代码，并进行防抖处理。
  • 当最后一个组件的 mounted() 触发时，就代表该路由下的所有组件都已经加载完毕，就可以在 this.$nextTick() 回调中计算渲染时间了。
  • 另外还要考虑到一个情况：不切换路由时，也会有变更组件的情况，此时就不应该在这些组件的 mounted() 中进行渲染时间计算。所以需要添加一个 needCalculateRenderTime 字段，当切换路由时将它设为 true，代表需要计算渲染时间。

import { report, PerformanceInfo } from './report.js';

function watchVueLoad({ vueCls, router } = {}) {
    if (!vueCls || !router) return;

    let startTime;
    router.beforeEach((to, from, next) => {
        // 给 router 新增一个字段，表示是否要计算渲染时间
        // 只有路由跳转才需要计算
        router.needCalculateRenderTime = true;
        startTime = performance.now();

        next();
    })

    let timer;
    vueCls.mixin({
        mounted() {
            if (!router.needCalculateRenderTime) return;

            // 整个组件的视图都被渲染之后才会运行nextTick中的代码
            this.$nextTick(() => {
                clearTimeout(timer);
                const now = performance.now();
                timer = setTimeout(() => {
                    router.needCalculateRenderTime = false;
                    report(new PerformanceInfo({
                        subType: "VueLoad",
                        value: now - startTime
                    }));
                }, 1000);
            })
        },
    })
}

export default watchVueLoad;
